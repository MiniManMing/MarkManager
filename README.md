## 1.7. 设计算法
### 1.7.1. NodeCopy
> 只说我程序中的一个亮点，排序等算法我认为并没有什么可说的，因为设计数据结构时，考虑到学生与成绩之间的关系为1:N的关系，故必须在每个新的学生节点后面挂载成绩链表，借鉴Java语言的new关键字，每new一个对象开辟一个内存空间的原则，进行设计。此函数可以复制一个新的且为空的Mark节点。
```
PMNODE MarkNodeCopy(PMNODE mhead){
    PMNODE h,a;
   if(mhead == NULL){
       printf("复制的链表是空的");
       return NULL;
   }else if(mhead->D==NULL){
       printf("复制的链表是空的");
       return NULL;
   }
   h=a=(PMNODE)malloc(sizeof(MNODE));
   a->D=(PMNODE)malloc(sizeof(MNODE));
   a=a->D;
   while (1){
       strcpy(a->mark.ClassName,mhead->mark.ClassName);
       a->mark.ClassMark=0;
       if(mhead->D!=NULL){
           mhead=mhead->D;
           a->D=(PMNODE)malloc(sizeof(MNODE));
           a=a->D;
       } else{
           a->D=NULL;
           break;
       }
   }
    return h;

}
```

void DeleteStudent(DN pHead){

        if(flag==d->student.tempId){

            DN h=d->UP;
            h->DOWN=t;
            t->UP=h;
            temp=t;

        d=d->DOWN;
    }

        temp->student.tempId--;
        temp=temp->DOWN;
    }

    OutputStudent(pHead);

}

void ScortByAVG(DN pHead) {
    DN p, q;//定义两个指针

    SNODE temp;

    for (p = pHead->DOWN; NULL != p; p = p->DOWN) {
        for (q = p->DOWN; NULL != q; q = q->DOWN) {
            if (p->Avg < q->Avg)//当前一个学生的语文成绩小于后一个学生的语文成绩时
            {
                printf("%.1f<%.1f\n",p->Avg,q->Avg);
                temp.student = p->student;//交换学生的位置
                p->student = q->student;
                q->student = temp.student;
            }
        }
    }
}
